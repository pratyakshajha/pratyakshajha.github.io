From May 1st, the government has opened several vaccination centers for people aged between 18 to 45. As the vaccines are scarce and logging in every time to check status is a cumbersome process, I have used open APIs provided by the government to keep track of the availability of vaccines in my home district of Nagpur.

import requests
import json

# Appointment availibility (18+)
url = "https://cdn-api.co-vin.in/api/v2/appointment/sessions/public/findByDistrict"
params = {"district_id": "365", "date": "07-05-2021"}
response = requests.get(url, params=params)
print("HTTP ", response.status_code)
data = response.json()
for session in data["sessions"]:
    if session["min_age_limit"] == 45:
        print(json.dumps(session, indent=2))

# Appointment availibility for (18+) for 7 days
url = "https://cdn-api.co-vin.in/api/v2/appointment/sessions/public/calendarByDistrict"
params = {"district_id": "365", "date": "07-05-2021"}
response = requests.get(url, params=params)
print("HTTP ", response.status_code)
data = response.json()
for center in data["centers"]:
    for session in center["sessions"]:
        if session["min_age_limit"] == 18 and session["vaccine"]=="COVISHIELD" and session["available_capacity"] > 0:
            print(json.dumps(session, indent=2))

# API for state and district lists
url = "https://cdn-api.co-vin.in/api/v2/admin/location/states"
response = requests.get(url)
print("HTTP ", response.status_code)
print(json.dumps(response.json(), indent=2))

state_id = "21"
url = "https://cdn-api.co-vin.in/api/v2/admin/location/districts/" + state_id
response = requests.get(url)
print("HTTP ", response.status_code)
print(json.dumps(response.json(), indent=2))
