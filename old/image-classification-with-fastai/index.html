Fastai is a deep learning library built with Pytorch for rapidly building neural networks. For image classification problems, I start with fastai to get a feel of the problem and set some realistic expectations.

This template acts as the first iteration of a solution. Customization and enhancements are done on top of this code to suit a problem statement.

from fastai.vision import *
from fastai.metrics import error_rate

path = untar_data(URLs.PETS)
path.ls()
path_anno = path/'annotations'
path_img = path/'images'
fnames = get_image_files(path_img)

np.random.seed(2)
pat = r'/([^/]+)_\d+.jpg$'
data = ImageDataBunch.from_name_re(path_img, fnames, pat, ds_tfms=get_transforms(), 
                                   size=224, bs=32).normalize(imagenet_stats)

data.show_batch(rows=3, figsize=(7,6))
print(data.classes)
print(len(data.classes))

# training
learn = cnn_learner(data, models.resnet34, metrics=error_rate)
learn.model
learn.fit_one_cycle(4)
learn.save('stage-1')

# interpreting results 
interp = ClassificationInterpretation.from_learner(learn)
losses,idxs = interp.top_losses()
len(data.valid_ds)==len(losses)==len(idxs)

interp.plot_top_losses(9, figsize=(15,11))
interp.plot_confusion_matrix(figsize=(12,12), dpi=60)
interp.most_confused(min_val=2)

# fine tuned training
learn.unfreeze()
learn.fit_one_cycle(1)
learn.load('stage-1')
learn.lr_find()
learn.recorder.plot()
learn.unfreeze()
learn.fit_one_cycle(2, max_lr=slice(1e-6,1e-4))
